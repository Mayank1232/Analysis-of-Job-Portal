start = time.time()

print("********Data collection process has been started********")

    while count <= page_count:

        driver.get(newpage)
        time.sleep(5)        
         # Find all the reviews:
        positions = driver.find_elements_by_xpath('//article[@class = "jobTuple bgWhite br4 mb-8"]')
